function quickSort(arr) {
	
	if (arr.length < 2)
		return arr;

	const pivot = arr[0];
	const less = [];
	const greater = [];

	for (let i = 1; i < arr.length; i++) {
		if (arr[i] < pivot) {
			less[less.length] = arr[i];
		} else {
			greater[greater.length] = arr[i];
		}
	}
	return quickSort(less).concat(pivot, quickSort(greater));
}
